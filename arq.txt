void buscarProfessor(Professor lista[], int tam) {
    int tipo, idBusca, encontrado = 0;
    char nomeBusca[50];
    printf("\nBuscar por: 1. ID | 2. Nome: ");
    scanf("%d", &tipo);
    limparBuffer();

    if (tipo == 1) {
        printf("Digite o ID: ");
        scanf("%d", &idBusca);
        for (int i = 0; i < tam; i++) {
            if (lista[i].id == idBusca) {
                printf("\nEncontrado: %s | Disciplina: %s | Idade: %d\n", lista[i].nome, lista[i].disciplina, lista[i].idade);
                encontrado = 1; break;
            }
        }
    } else {
        printf("Digite o Nome: ");
        fgets(nomeBusca, 50, stdin);
        nomeBusca[strcspn(nomeBusca, "\n")] = 0;
        
        int matches[MAX], mCont = 0;
        for (int i = 0; i < tam; i++) {
            if (strstr(lista[i].nome, nomeBusca) != NULL) matches[mCont++] = i;
        }

        if (mCont > 1) {
            printf("\nVarios professores encontrados. Qual disciplina voce busca?\n");
            for(int i=0; i<mCont; i++) printf("[%d] ID: %d | Nome: %s | Materia: %s\n", i+1, lista[matches[i]].id, lista[matches[i]].nome, lista[matches[i]].disciplina);
            printf("Escolha o numero: ");
            int sel; scanf("%d", &sel);
            if(sel > 0 && sel <= mCont) {
                int idx = matches[sel-1];
                printf("\nSelecionado: %s | Disciplina: %s\n", lista[idx].nome, lista[idx].disciplina);
                encontrado = 1;
            }
        } else if (mCont == 1) {
            printf("\nID: %d | Nome: %s | Disciplina: %s\n", lista[matches[0]].id, lista[matches[0]].nome, lista[matches[0]].disciplina);
            encontrado = 1;
        }
    }
    if (!encontrado) printf("\nNenhum registro encontrado.\n");
}

void excluirProfessor(Professor lista[], int *tam) {
    int id; printf("ID para excluir: "); scanf("%d", &id);
    for(int i=0; i<*tam; i++) {
        if(lista[i].id == id) {
            for(int j=i; j<*tam-1; j++) lista[j] = lista[j+1];
            (*tam)--; printf("Excluido.\n"); return;
        }
    }
}
