#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX 100

typedef struct {
    int id;
    char tipo[40];
    char nome[100];
    int idade;
    char campo1[100];
    char campo2[100];
    char campo3[100];
    char email[100];
} Registro;

void limparBuffer() {
    int c;
    while ((c = getchar()) != '\n' && c != EOF);
}

void removerEnter(char texto[]) {
    texto[strcspn(texto, "\n")] = '\0';
}

void salvar(char nomeArquivo[], Registro lista[], int total) {
    FILE *arquivo = fopen(nomeArquivo, "wb");

    if (arquivo == NULL) {
        printf("\nErro ao salvar arquivo.\n");
        return;
    }

    fwrite(&total, sizeof(int), 1, arquivo);
    fwrite(lista, sizeof(Registro), total, arquivo);

    fclose(arquivo);
}

int carregar(char nomeArquivo[], Registro lista[]) {
    FILE *arquivo = fopen(nomeArquivo, "rb");
    int total = 0;

    if (arquivo == NULL) {
        return 0;
    }

    fread(&total, sizeof(int), 1, arquivo);
    fread(lista, sizeof(Registro), total, arquivo);

    fclose(arquivo);
    return total;
}

int existeId(Registro lista[], int total, int id) {
    int i;

    for (i = 0; i < total; i++) {
        if (lista[i].id == id) {
            return 1;
        }
    }

    return 0;
}

int gerarId(Registro lista[], int total) {
    int maior = 0;
    int i;

    for (i = 0; i < total; i++) {
        if (lista[i].id > maior) {
            maior = lista[i].id;
        }
    }

    return maior + 1;
}

void mostrarRegistro(Registro r, char c1[], char c2[], char c3[]) {
    printf("\nID: %d\n", r.id);
    printf("Tipo: %s\n", r.tipo);
    printf("Nome: %s\n", r.nome);
    printf("Idade: %d\n", r.idade);
    printf("%s: %s\n", c1, r.campo1);
    printf("%s: %s\n", c2, r.campo2);
    printf("%s: %s\n", c3, r.campo3);
    printf("Email: %s\n", r.email);
}

void listar(Registro lista[], int total, char c1[], char c2[], char c3[]) {
    int i;

    if (total == 0) {
        printf("\nNenhum registro cadastrado.\n");
        return;
    }

    for (i = 0; i < total; i++) {
        mostrarRegistro(lista[i], c1, c2, c3);
        printf("-----------------------------\n");
    }
}

void cadastrar(Registro lista[], int *total, char arquivo[], char c1[], char c2[], char c3[]) {
    Registro novo;
    int i;

    if (*total >= MAX) {
        printf("\nLimite de cadastros atingido.\n");
        return;
    }

    novo.id = gerarId(lista, *total);

    printf("\nTipo: ");
    fgets(novo.tipo, 40, stdin);
    removerEnter(novo.tipo);

    printf("Nome: ");
    fgets(novo.nome, 100, stdin);
    removerEnter(novo.nome);

    for (i = 0; i < *total; i++) {
        if (strcmp(lista[i].nome, novo.nome) == 0) {
            printf("\nAviso: ja existe alguem com esse nome.\n");
            printf("ID: %d | Tipo: %s | %s: %s | %s: %s\n",
                   lista[i].id, lista[i].tipo, c1, lista[i].campo1, c2, lista[i].campo2);
        }
    }

    printf("Idade: ");
    scanf("%d", &novo.idade);
    limparBuffer();

    printf("%s: ", c1);
    fgets(novo.campo1, 100, stdin);
    removerEnter(novo.campo1);

    printf("%s: ", c2);
    fgets(novo.campo2, 100, stdin);
    removerEnter(novo.campo2);

    printf("%s: ", c3);
    fgets(novo.campo3, 100, stdin);
    removerEnter(novo.campo3);

    printf("Email: ");
    fgets(novo.email, 100, stdin);
    removerEnter(novo.email);

    lista[*total] = novo;
    (*total)++;

    salvar(arquivo, lista, *total);

    printf("\nCadastro realizado com sucesso! ID gerado: %d\n", novo.id);
}

void buscar(Registro lista[], int total, char c1[], char c2[], char c3[]) {
    int opcao, id, i, encontrados[MAX], qtd = 0, escolha;
    char nome[100];

    printf("\nBuscar por:\n");
    printf("1 - ID\n");
    printf("2 - Nome\n");
    printf("Escolha: ");
    scanf("%d", &opcao);
    limparBuffer();

    if (opcao == 1) {
        printf("Digite o ID: ");
        scanf("%d", &id);

        for (i = 0; i < total; i++) {
            if (lista[i].id == id) {
                printf("\nRegistro encontrado:\n");
                mostrarRegistro(lista[i], c1, c2, c3);
                return;
            }
        }

        printf("\nRegistro nao encontrado.\n");
    } else if (opcao == 2) {
        printf("Digite o nome: ");
        fgets(nome, 100, stdin);
        removerEnter(nome);

        for (i = 0; i < total; i++) {
            if (strstr(lista[i].nome, nome) != NULL) {
                encontrados[qtd] = i;
                qtd++;
            }
        }

        if (qtd == 0) {
            printf("\nNenhum registro encontrado.\n");
        } else if (qtd == 1) {
            mostrarRegistro(lista[encontrados[0]], c1, c2, c3);
        } else {
            printf("\nForam encontrados %d registros parecidos.\n", qtd);

            for (i = 0; i < qtd; i++) {
                int pos = encontrados[i];

                printf("\n[%d]\n", i + 1);
                printf("ID: %d\n", lista[pos].id);
                printf("Tipo: %s\n", lista[pos].tipo);
                printf("Nome: %s\n", lista[pos].nome);
                printf("%s: %s\n", c1, lista[pos].campo1);
                printf("%s: %s\n", c2, lista[pos].campo2);
                printf("Email: %s\n", lista[pos].email);
            }

            printf("\nEscolha o numero correto: ");
            scanf("%d", &escolha);

            if (escolha >= 1 && escolha <= qtd) {
                mostrarRegistro(lista[encontrados[escolha - 1]], c1, c2, c3);
            } else {
                printf("\nOpcao invalida.\n");
            }
        }
    } else {
        printf("\nOpcao invalida.\n");
    }
}

void editar(Registro lista[], int total, char arquivo[], char c1[], char c2[], char c3[]) {
    int id, i;

    printf("\nDigite o ID para editar: ");
    scanf("%d", &id);
    limparBuffer();

    for (i = 0; i < total; i++) {
        if (lista[i].id == id) {
            printf("\nEditando: %s\n", lista[i].nome);

            printf("Novo tipo: ");
            fgets(lista[i].tipo, 40, stdin);
            removerEnter(lista[i].tipo);

            printf("Novo nome: ");
            fgets(lista[i].nome, 100, stdin);
            removerEnter(lista[i].nome);

            printf("Nova idade: ");
            scanf("%d", &lista[i].idade);
            limparBuffer();

            printf("Novo %s: ", c1);
            fgets(lista[i].campo1, 100, stdin);
            removerEnter(lista[i].campo1);

            printf("Novo %s: ", c2);
            fgets(lista[i].campo2, 100, stdin);
            removerEnter(lista[i].campo2);

            printf("Novo %s: ", c3);
            fgets(lista[i].campo3, 100, stdin);
            removerEnter(lista[i].campo3);

            printf("Novo email: ");
            fgets(lista[i].email, 100, stdin);
            removerEnter(lista[i].email);

            salvar(arquivo, lista, total);

            printf("\nRegistro editado com sucesso.\n");
            return;
        }
    }

    printf("\nRegistro nao encontrado.\n");
}

void excluir(Registro lista[], int *total, char arquivo[]) {
    int id, i, j;

    printf("\nDigite o ID para excluir: ");
    scanf("%d", &id);

    for (i = 0; i < *total; i++) {
        if (lista[i].id == id) {
            printf("\nExcluindo: %s\n", lista[i].nome);

            for (j = i; j < *total - 1; j++) {
                lista[j] = lista[j + 1];
            }

            (*total)--;
            salvar(arquivo, lista, *total);

            printf("\nRegistro excluido com sucesso.\n");
            return;
        }
    }

    printf("\nRegistro nao encontrado.\n");
}

void preencherIniciais(Registro lista[], int *total, Registro iniciais[], int qtd, char arquivo[]) {
    int i;

    if (*total == 0) {
        for (i = 0; i < qtd; i++) {
            lista[i] = iniciais[i];
        }

        *total = qtd;
        salvar(arquivo, lista, *total);
    }
}

void menuBanco(char titulo[], char arquivo[], Registro lista[], int *total, char c1[], char c2[], char c3[]) {
    int opcao;

    do {
        printf("\n====================================\n");
        printf("BANCO DE DADOS: %s\n", titulo);
        printf("====================================\n");
        printf("1 - Cadastrar\n");
        printf("2 - Listar todos\n");
        printf("3 - Buscar\n");
        printf("4 - Editar\n");
        printf("5 - Excluir\n");
        printf("0 - Voltar\n");
        printf("Escolha uma opcao: ");
        scanf("%d", &opcao);
        limparBuffer();

        switch (opcao) {
            case 1:
                cadastrar(lista, total, arquivo, c1, c2, c3);
                break;
            case 2:
                listar(lista, *total, c1, c2, c3);
                break;
            case 3:
                buscar(lista, *total, c1, c2, c3);
                break;
            case 4:
                editar(lista, *total, arquivo, c1, c2, c3);
                break;
            case 5:
                excluir(lista, total, arquivo);
                break;
            case 0:
                printf("\nVoltando...\n");
                break;
            default:
                printf("\nOpcao invalida.\n");
        }

    } while (opcao != 0);
}
