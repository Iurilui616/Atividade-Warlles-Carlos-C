void buscarProfessor(Professor lista[], int tam) {
    int tipo, idBusca, encontrado = 0;
    char nomeBusca[50];

    printf("\nBuscar por: 1. ID | 2. Nome: ");
    scanf("%d", &tipo);
    getchar();

    if (tipo == 1) {
        printf("Digite o ID: ");
        scanf("%d", &idBusca);
        for (int i = 0; i < tam; i++) {
            if (lista[i].id == idBusca) {
                printf("\nEncontrado: %s | Disciplina: %s | %s\n", lista[i].nome, lista[i].disciplina, lista[i].turmas);
                encontrado = 1;
                break;
            }
        }
    } else {
        printf("Digite o Nome (ou parte dele): ");
        fgets(nomeBusca, 50, stdin);
        nomeBusca[strcspn(nomeBusca, "\n")] = 0;

        for (int i = 0; i < tam; i++) {
            if (strstr(lista[i].nome, nomeBusca) != NULL) {
                printf("\nID: %d | Nome: %s | Disciplina: %s\n", lista[i].id, lista[i].nome, lista[i].disciplina);
                encontrado = 1;
            }
        }
    }
    if (!encontrado) printf("\nNenhum professor encontrado.\n");
}
