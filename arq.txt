void limparNomeArquivo(char texto[]) {
    int i, j = 0;
    char limpo[120];

    for (i = 0; texto[i] != '\0' && j < 110; i++) {
        unsigned char c = (unsigned char)texto[i];

        if ((c >= 'A' && c <= 'Z') ||
            (c >= 'a' && c <= 'z') ||
            (c >= '0' && c <= '9') ||
            c == '_' || c == '-') {
            limpo[j++] = (char)c;
        } else if (c == ' ') {
            limpo[j++] = '_';
        }
    }

    limpo[j] = '\0';

    if (strlen(limpo) == 0) {
        strcpy(limpo, "principal");
    }

    strcpy(texto, limpo);
}

void montarPastaBanco(char destino[], int modelo, char nomeBanco[]) {
    char nomeLimpo[120];

    strcpy(nomeLimpo, nomeBanco);
    limparNomeArquivo(nomeLimpo);

    sprintf(destino, "%s\\%s_%s", pastaDados, nomesArquivosBase[modelo], nomeLimpo);
}

void montarArquivoBanco(char destino[], int modelo, char nomeBanco[]) {
    char pasta[360];

    montarPastaBanco(pasta, modelo, nomeBanco);
    sprintf(destino, "%s\\registros.dat", pasta);
}

int arquivoExiste(char caminho[]) {
    FILE *arquivo = fopen(caminho, "rb");

    if (arquivo == NULL) {
        return 0;
    }

    fclose(arquivo);
    return 1;
}

void salvar(char nomeArquivo[], Registro lista[], int total) {
    FILE *arquivo = fopen(nomeArquivo, "wb");

    if (arquivo == NULL) {
        MessageBoxA(hJanela, "Erro ao salvar. Verifique se a pasta do banco existe.", "Erro", MB_ICONERROR);
        return;
    }

    fwrite(&total, sizeof(int), 1, arquivo);

    if (total > 0) {
        fwrite(lista, sizeof(Registro), total, arquivo);
    }

    fclose(arquivo);
}

int carregar(char nomeArquivo[], Registro lista[]) {
    FILE *arquivo = fopen(nomeArquivo, "rb");
    int total = 0;

    if (arquivo == NULL) {
        return 0;
    }

    if (fread(&total, sizeof(int), 1, arquivo) != 1) {
        fclose(arquivo);
        return 0;
    }

    if (total < 0 || total > MAX) {
        fclose(arquivo);
        return 0;
    }

    if (total > 0) {
        fread(lista, sizeof(Registro), total, arquivo);
    }

    fclose(arquivo);
    return total;
}

void salvarInfoBanco(Banco *b, int modelo, char nomeBanco[]) {

    char caminho[500];

    sprintf(caminho, "%s\\info.txt", b->pastaBanco);

    f = fopen(caminho, "w");

    if (f != NULL) {
        fprintf(f, "Nome do banco: %s\n", nomeBanco);
        fprintf(f, "Modelo: %s\n", nomesModelos[modelo]);
        fprintf(f, "Total de registros: %d\n", b->total);
        fprintf(f, "Arquivo de dados: registros.dat\n");
        fprintf(f, "Pasta: %s\n", b->pastaBanco);
        fclose(f);
    }
}

int gerarId(Registro lista[], int total) {
    int maior = 0;
    int i;

    for (i = 0; i < total; i++) {
        if (lista[i].id > maior) {
            maior = lista[i].id;
        }
    }

    return maior + 1;
}

void copiarRegistros(Registro destino[], int *total, Registro origem[], int qtd) {
    int i;

    for (i = 0; i < qtd && i < MAX; i++) {
        destino[i] = origem[i];
    }

    *total = qtd;
}

int contemTexto(const char *texto, const char *filtro) {
    char a[300];
    char b[120];
    int i;

    if (filtro == NULL || strlen(filtro) == 0) {
        return 1;
    }

    strncpy(a, texto, sizeof(a) - 1);
    a[sizeof(a) - 1] = '\0';

    strncpy(b, filtro, sizeof(b) - 1);
    b[sizeof(b) - 1] = '\0';

    for (i = 0; a[i]; i++) a[i] = (char)tolower((unsigned char)a[i]);
    for (i = 0; b[i]; i++) b[i] = (char)tolower((unsigned char)b[i]);

    return strstr(a, b) != NULL;
}

void atualizarLista(const char *filtro) {
    int i;
    char linha[520];
    Banco *b = &bancos[bancoAtual];

    SendMessageA(hLista, LB_RESETCONTENT, 0, 0);

    for (i = 0; i < b->total; i++) {
        char idTxt[30];
        int mostrar = 0;

        sprintf(idTxt, "%d", b->lista[i].id);

        if (contemTexto(idTxt, filtro) ||
            contemTexto(b->lista[i].nome, filtro) ||
            contemTexto(b->lista[i].tipo, filtro) ||
            contemTexto(b->lista[i].campo1, filtro) ||
            contemTexto(b->lista[i].campo2, filtro) ||
            contemTexto(b->lista[i].campo3, filtro) ||
            contemTexto(b->lista[i].email, filtro)) {
            mostrar = 1;
        }

        if (mostrar) {
            sprintf(linha, "#%03d | %-11s | %-24s | %s: %-18s | %s",
                    b->lista[i].id,
                    b->lista[i].tipo,
                    b->lista[i].nome,
                    b->c1,
                    b->lista[i].campo1,
                    b->lista[i].email);

            int pos = (int)SendMessageA(hLista, LB_ADDSTRING, 0, (LPARAM)linha);
            SendMessageA(hLista, LB_SETITEMDATA, pos, i);
        }
    }

    sprintf(linha, "Banco: %s | Registros: %d | Pasta: %s", b->titulo, b->total, b->pastaBanco);
    setStatus(linha);
    InvalidateRect(hJanela, NULL, FALSE);
}

int indicePorId(Banco *b, int id) {
    int i;

    for (i = 0; i < b->total; i++) {
        if (b->lista[i].id == id) {
            return i;
        }
    }

    return -1;
}

void buscarRegistros() {
    char filtro[120];
    getTexto(hBusca, filtro, 120);
    atualizarLista(filtro);
}

void selecionarDaLista() {
    int sel = (int)SendMessageA(hLista, LB_GETCURSEL, 0, 0);
    int indice;
    Banco *b = &bancos[bancoAtual];

    if (sel == LB_ERR) {
        return;
    }

    indice = (int)SendMessageA(hLista, LB_GETITEMDATA, sel, 0);

    if (indice >= 0 && indice < b->total) {
        mostrarRegistroNosCampos(b->lista[indice]);
        setStatus("Registro selecionado. Edite os campos e clique em Salvar cadastro.");
    }
}

void salvarRegistroDaTela() {
    Banco *b = &bancos[bancoAtual];
    Registro r;
    char txt[120];
    int pos;

    getTexto(hId, txt, 30);
    r.id = atoi(txt);

    if (r.id <= 0) {
        r.id = gerarId(b->lista, b->total);
    }

    getTexto(hTipo, r.tipo, 40);
    getTexto(hNome, r.nome, 100);
    getTexto(hIdade, txt, 30);
    r.idade = atoi(txt);
    getTexto(hCampo1, r.campo1, 100);
    getTexto(hCampo2, r.campo2, 100);
    getTexto(hCampo3, r.campo3, 100);
    getTexto(hEmail, r.email, 100);

    if (strlen(r.tipo) == 0 || strlen(r.nome) == 0) {
        MessageBoxA(hJanela, "Preencha pelo menos Tipo e Nome.", "Campos obrigatorios", MB_ICONWARNING);
        return;
    }

    pos = indicePorId(b, r.id);

    if (pos >= 0) {
        b->lista[pos] = r;
        salvar(b->arquivo, b->lista, b->total);
        salvarInfoBanco(b, bancoAtual, b->titulo);
        atualizarLista("");
        MessageBoxA(hJanela, "Registro editado e salvo com sucesso.", "Sucesso", MB_ICONINFORMATION);
        return;
    }

    if (b->total >= MAX) {
        MessageBoxA(hJanela, "Limite de cadastros atingido.", "Aviso", MB_ICONWARNING);
        return;
    }

    {
        int i;
        char aviso[1000];
        strcpy(aviso, "");

        for (i = 0; i < b->total; i++) {
            if (strcmp(b->lista[i].nome, r.nome) == 0) {
                char temp[220];

                sprintf(temp, "Nome repetido encontrado:\nID: %d | Tipo: %s | %s: %s\n\n",
                        b->lista[i].id,
                        b->lista[i].tipo,
                        b->c1,
                        b->lista[i].campo1);

                strcat(aviso, temp);
            }
        }

        if (strlen(aviso) > 0) {
            strcat(aviso, "O cadastro sera salvo mesmo assim, com ID diferente.");
            MessageBoxA(hJanela, aviso, "Aviso", MB_ICONWARNING);
        }
    }

    b->lista[b->total] = r;
    b->total++;

    salvar(b->arquivo, b->lista, b->total);
    salvarInfoBanco(b, bancoAtual, b->titulo);
    atualizarLista("");
    limparCampos();

    MessageBoxA(hJanela, "Cadastro salvo com sucesso.", "Sucesso", MB_ICONINFORMATION);
}

void excluirSelecionado() {
    Banco *b = &bancos[bancoAtual];
    char txt[30];
    int id, pos, i;

    getTexto(hId, txt, 30);
    id = atoi(txt);
    pos = indicePorId(b, id);

    if (pos < 0) {
        MessageBoxA(hJanela, "Selecione um registro ou digite um ID valido.", "Registro nao encontrado", MB_ICONWARNING);
        return;
    }

    if (MessageBoxA(hJanela, "Deseja realmente excluir este registro?", "Confirmar exclusao",
                    MB_YESNO | MB_ICONQUESTION) != IDYES) {
        return;
    }

    for (i = pos; i < b->total - 1; i++) {
        b->lista[i] = b->lista[i + 1];
    }

    b->total--;

    salvar(b->arquivo, b->lista, b->total);
    salvarInfoBanco(b, bancoAtual, b->titulo);
    atualizarLista("");
    limparCampos();

    MessageBoxA(hJanela, "Registro excluido com sucesso.", "Sucesso", MB_ICONINFORMATION);
}

void abrirBancoPorNome(int criarNovo) {
    int modelo;
    char nomeBanco[120];
    char caminho[360];
    char pasta[360];
    char msg[750];
    Banco *b;

    modelo = (int)SendMessageA(hComboModelo, CB_GETCURSEL, 0, 0);
    if (modelo < 0 || modelo > 3) {
        modelo = 0;
    }

    getTexto(hNomeBanco, nomeBanco, 120);
    limparNomeArquivo(nomeBanco);

    montarPastaBanco(pasta, modelo, nomeBanco);
    montarArquivoBanco(caminho, modelo, nomeBanco);

    bancoAtual = modelo;
    b = &bancos[bancoAtual];

    configurarRotulosBanco(b, modelo);
    strcpy(b->pastaBanco, pasta);
    strcpy(b->arquivo, caminho);

    if (criarNovo) {
        criarPastaSeNaoExistir(pasta);

        if (arquivoExiste(caminho)) {
            if (MessageBoxA(hJanela, "Ja existe um banco com esse nome. Deseja substituir?", "Banco ja existe",
                            MB_YESNO | MB_ICONQUESTION) != IDYES) {
                return;
            }
        }

        carregarModeloInicial(b, modelo);
        salvar(b->arquivo, b->lista, b->total);
        salvarInfoBanco(b, modelo, nomeBanco);

        SendMessageA(hComboBanco, CB_SETCURSEL, modelo, 0);
        atualizarBancoAtual();

        sprintf(msg, "Banco criado com sucesso!\n\nPasta:\n%s\n\nArquivo:\n%s", b->pastaBanco, b->arquivo);
        MessageBoxA(hJanela, msg, "Banco criado", MB_ICONINFORMATION);
    } else {
        if (!arquivoExiste(caminho)) {
            MessageBoxA(hJanela, "Esse banco ainda nao existe. Confira o nome ou clique em Criar banco.",
                        "Banco nao encontrado", MB_ICONWARNING);
            return;
        }

        b->total = carregar(b->arquivo, b->lista);

        SendMessageA(hComboBanco, CB_SETCURSEL, modelo, 0);
        atualizarBancoAtual();

        sprintf(msg, "Banco aberto com sucesso!\n\nPasta:\n%s", b->pastaBanco);
        MessageBoxA(hJanela, msg, "Banco aberto", MB_ICONINFORMATION);
    }
}

}

void inicializarBancos() {
    int i;
    char pasta[360];
    char arquivo[360];

    criarPastaDados();

    for (i = 0; i < 4; i++) {
        configurarRotulosBanco(&bancos[i], i);

        montarPastaBanco(pasta, i, "principal");
        criarPastaSeNaoExistir(pasta);

        montarArquivoBanco(arquivo, i, "principal");

        strcpy(bancos[i].pastaBanco, pasta);
        strcpy(bancos[i].arquivo, arquivo);

        bancos[i].total = carregar(bancos[i].arquivo, bancos[i].lista);

        if (bancos[i].total == 0) {
            carregarModeloInicial(&bancos[i], i);
            salvar(bancos[i].arquivo, bancos[i].lista, bancos[i].total);
            salvarInfoBanco(&bancos[i], i, "principal");
        }
    }
}

void desenharTexto(HDC hdc, const char *texto, int x, int y, int w, int h, HFONT fonte, COLORREF cor, UINT formato) {
    RECT r;

    r.left = x;
    r.top = y;
    r.right = x + w;
    r.bottom = y + h;

    SetBkMode(hdc, TRANSPARENT);
    SetTextColor(hdc, cor);
    SelectObject(hdc, fonte);
    DrawTextA(hdc, texto, -1, &r, formato);
}

void desenharCard(HDC hdc, int x, int y, int w, int h) {

}

void desenharBotao(LPDRAWITEMSTRUCT d) {
    int tipo = (int)GetWindowLongPtrA(d->hwndItem, GWLP_USERDATA);
    COLORREF cor = COR_AZUL;
    COLORREF corTexto = RGB(255,255,255);
    char texto[120];
    HBRUSH brush;
    HPEN pen;
    HGDIOBJ oldBrush, oldPen;
    RECT r;

    if (tipo == BTN_AZUL) {
        cor = COR_AZUL;
        corTexto = RGB(255,255,255);
    } else if (tipo == BTN_VERDE) {
        cor = COR_VERDE;
        corTexto = RGB(255,255,255);
    } else if (tipo == BTN_VERMELHO) {
        cor = COR_VERMELHO;
        corTexto = RGB(255,255,255);
    } else if (tipo == BTN_ESCURO) {
        cor = RGB(30, 64, 120);
        corTexto = RGB(255,255,255);
    } else {
        cor = RGB(226, 235, 247);
        corTexto = RGB(27, 65, 120);
    }

    if (d->itemState & ODS_SELECTED) {
        if (tipo == BTN_CLARO) {
            cor = RGB(210, 224, 245);
        } else {
            cor = RGB(GetRValue(cor) > 25 ? GetRValue(cor) - 20 : 0,
                      GetGValue(cor) > 25 ? GetGValue(cor) - 20 : 0,
                      GetBValue(cor) > 25 ? GetBValue(cor) - 20 : 0);
        }
    }

    GetWindowTextA(d->hwndItem, texto, sizeof(texto));

    brush = CreateSolidBrush(cor);
    pen = CreatePen(PS_SOLID, 1, cor);

    oldBrush = SelectObject(d->hDC, brush);
    oldPen = SelectObject(d->hDC, pen);

    RoundRect(d->hDC, d->rcItem.left, d->rcItem.top, d->rcItem.right, d->rcItem.bottom, 14, 14);

    SelectObject(d->hDC, oldBrush);
    SelectObject(d->hDC, oldPen);

    DeleteObject(brush);
    DeleteObject(pen);

    r = d->rcItem;

    SetBkMode(d->hDC, TRANSPARENT);
    SetTextColor(d->hDC, corTexto);
    SelectObject(d->hDC, fonteBotao);
    DrawTextA(d->hDC, texto, -1, &r, DT_CENTER | DT_VCENTER | DT_SINGLELINE);
}

LRESULT CALLBACK JanelaProc(HWND hwnd, UINT msg, WPARAM wParam, LPARAM lParam) {
    switch (msg) {
        case WM_CREATE:
            hJanela = hwnd;
            criarFontes();
            criarBrushes();
            inicializarBancos();
            criarInterface(hwnd);
            atualizarBancoAtual();
            return 0;

        case WM_PAINT: {
            PAINTSTRUCT ps;
            HDC hdc = BeginPaint(hwnd, &ps);
            desenharInterface(hdc);
            EndPaint(hwnd, &ps);
            return 0;
        }

        case WM_DRAWITEM:
            if (((LPDRAWITEMSTRUCT)lParam)->CtlType == ODT_BUTTON) {
                desenharBotao((LPDRAWITEMSTRUCT)lParam);
                return TRUE;
            }
            break;

        case WM_CTLCOLORSTATIC: {
            HDC hdc = (HDC)wParam;
            HWND hCtrl = (HWND)lParam;
            RECT tela;
            POINT p;

            SetBkMode(hdc, TRANSPARENT);

            if (hCtrl == hStatus) {
                SetTextColor(hdc, COR_MUTED);
                return (LRESULT)brushCard;
            }

            GetWindowRect(hCtrl, &tela);
            p.x = tela.left;
            p.y = tela.top;
            ScreenToClient(hwnd, &p);

            if (p.x < 260) {
                SetTextColor(hdc, RGB(220, 235, 255));
                return (LRESULT)brushSidebar;
            }

            SetTextColor(hdc, COR_MUTED);
            return (LRESULT)brushCard;
        }

        case WM_CTLCOLOREDIT: {
            HDC hdc = (HDC)wParam;
            SetBkColor(hdc, COR_INPUT);
            SetTextColor(hdc, COR_TEXTO);
            return (LRESULT)brushInput;
        }

        case WM_CTLCOLORLISTBOX: {
            HDC hdc = (HDC)wParam;
            SetBkColor(hdc, RGB(255,255,255));
            SetTextColor(hdc, COR_TEXTO);
            return (LRESULT)brushCard;
        }

        case WM_COMMAND:
            switch (LOWORD(wParam)) {
                case ID_COMBO_BANCO:
                    if (HIWORD(wParam) == CBN_SELCHANGE) {
                        bancoAtual = (int)SendMessageA(hComboBanco, CB_GETCURSEL, 0, 0);
                        if (bancoAtual < 0 || bancoAtual > 3) bancoAtual = 0;
                        SendMessageA(hComboModelo, CB_SETCURSEL, bancoAtual, 0);
                        atualizarBancoAtual();
                    }
                    break;

                case ID_LISTA:
                    if (HIWORD(wParam) == LBN_SELCHANGE) {
                        selecionarDaLista();
                    }
                    break;

                case ID_BUSCA:
                    if (HIWORD(wParam) == EN_CHANGE) {
                        buscarRegistros();
                    }
                    break;

                case ID_BTN_BUSCAR:
                    buscarRegistros();
                    break;

                case ID_BTN_LISTAR:
                    setTexto(hBusca, "");
                    atualizarLista("");
                    break;

                case ID_BTN_NOVO:
                    limparCampos();
                    break;

                case ID_BTN_SALVAR:
                    salvarRegistroDaTela();
                    break;

                case ID_BTN_EXCLUIR:
                    excluirSelecionado();
                    break;

                case ID_BTN_LIMPAR:
                    limparCampos();
                    break;

                case ID_BTN_CRIAR_BANCO:
                    abrirBancoPorNome(1);
                    break;

                case ID_BTN_ENTRAR_BANCO:
                    abrirBancoPorNome(0);
                    break;

                case ID_BTN_ABRIR_PASTA:
                    ShellExecuteA(hwnd, "open", bancos[bancoAtual].pastaBanco, NULL, NULL, SW_SHOWNORMAL);
                    break;
            }
            return 0;

        case WM_DESTROY:
            destruirObjetos();
            PostQuitMessage(0);
            return 0;
    }

    return DefWindowProcA(hwnd, msg, wParam, lParam);
}

int WINAPI WinMain(HINSTANCE hInstance, HINSTANCE hPrevInstance, LPSTR lpCmdLine, int nCmdShow) {
    WNDCLASSA wc;
    MSG msg;

    (void)hPrevInstance;
    (void)lpCmdLine;

    hInst = hInstance;

    memset(&wc, 0, sizeof(WNDCLASSA));

    wc.lpfnWndProc = JanelaProc;
    wc.hInstance = hInstance;
    wc.lpszClassName = "SistemaBancoBonitoEstavel";
    wc.hCursor = LoadCursor(NULL, IDC_ARROW);
    wc.hIcon = LoadIcon(NULL, IDI_APPLICATION);
    wc.hbrBackground = (HBRUSH)(COLOR_WINDOW + 1);

    if (!RegisterClassA(&wc)) {
        MessageBoxA(NULL, "Erro ao registrar a janela.", "Erro", MB_ICONERROR);
        return 0;
    }

    hJanela = CreateWindowExA(
        0,
        "SistemaBancoBonitoEstavel",
        "Sistema de Banco de Dados",
        WS_OVERLAPPED | WS_CAPTION | WS_SYSMENU | WS_MINIMIZEBOX,
        CW_USEDEFAULT,
        CW_USEDEFAULT,
        1065,
        875,
        NULL,
        NULL,
        hInstance,
        NULL
    );

    if (hJanela == NULL) {
        MessageBoxA(NULL, "Erro ao criar a janela.", "Erro", MB_ICONERROR);
        return 0;
    }

    ShowWindow(hJanela, nCmdShow);
    UpdateWindow(hJanela);

    while (GetMessageA(&msg, NULL, 0, 0)) {
        TranslateMessage(&msg);
        DispatchMessageA(&msg);
    }

    return (int)msg.wParam;
}
